SQL:

SELECT 
        p.id AS person_id,
        CONCAT(p.fname, ' ', p.lname) AS person_name,
        COUNT(DISTINCT o.id) AS order_count
    FROM 
        people p
    LEFT JOIN 
        orders o ON p.id = o.person
    GROUP BY 
        p.id, p.fname, p.lname;
